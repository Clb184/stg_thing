Types:
-Config -> Configuration that game retrieves at start
{
-playername (string)
-lives (int)
-bombs (int)
-sndvol (int 0-100)
-snden (bool)
-bgmvol (int 0-100)
-bgmen (bool)
-windowed (int)
-resolution (int)
-fetch_url (string)

}
-ScoreTable -> Score table compiling information about past runs
{
	int number_of_entries
	ScoreInfo* score_entry

}

-ScoreEntry -> Entry includes Name, Score, Date, ShotName
Retrieved from following JSON:
{
	"name" : "",
	"score" : 00000,
	"date" : "01/01/2026",
	"shotname" : "Shot sample"
}

-MusicEntry -> Music Entry fed to the Music Control, composed by:
{
	"path" : "", // Path of music file, mandatory, if failed to add, no music is played
	"loop" : true, // True by default
	"beginloop" : "0", // Start of track in samples by default
	"endloop" : "20000" // End of track in samples by default
}

-SoundEntry -> Sound Entry fed to the Sound Control, composed by:
{
	"path" : "sound.ogg", // 
	"nbuffers" : 2 // Number of buffers, default is 1
}

-Sprite -> Basic sprite info
{
	int tex_id
	float x, y
	float w, h
	uint32_t color
	float x1, y1, x2, y2
	int vb
}

-SpriteBatchBit -> Basic Batched sprite info
{
	float x, y
	float w, h
	uint32_t color
	float x1, y1, x2, y2
}

-TaskStage -> Task related to stage control logic
{
	vm task
	float gr[8];
	vm_interrupt interrupts[4]
}

-TaskGUI -> Task for drawing the in-game GUI
{
	vm task
	float x, y
	float gr[8]
	vm_interrupt interrupts[4]
}

-TaskEnemy -> Task for enemies
{
	vm task
	vm anmtask
	float x, y
	int enm_flags
	int hp
	int scoreval
	float gr[8]
	int atk_ids[4]
	// TaskEnemy* parent -> to implement later
	// TaskEnemy child[4]
}

-TaskPlayer -> Task that handles player states
{
	vm task
	float x, y
	int state_flags
	float gr[4]
	vm_function on_spawn
	vm_function on_death

	vm_function bomb_proc
}

-TaskPlayerShot -> Task that handles a player shot
{
	vm task
	float x, y
	int damage -> This is always positive, else will be 0
	float gr[4]
}

-DamageArea -> Describes a circular damage área
{
	float x, y
	int damage_cnt -> If negative, heals enemies and Boss
	
}

-HitboxArea -> Describes an enemy circular hitbox
{
	float x, y
	float r -> radius to compare to
}

-TaskBackground2D -> Object show as Surface background
{
	uint32 color
}

-TaskBackground3D -> Object shown in 3D background
{
	vm task
	float x, y, z
	float pitch, yaw, roll
	uint32 color
}

-TaskCutIn -> Big Sprite
{
	vm task
	float x, y
	uint32 color
	float gr[4]
}

-TaskText -> Data related to text rendering, 
{
	vm task
	float x, y
	uint32 color
	int font_id
	string text
}

-TaskEffect -> Effects, that's it
